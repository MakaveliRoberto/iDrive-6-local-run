! Configuration data used during the region detection phase.

:DETECT
    !----------------------------------------------------------------
    ! Header fields

    :HEADER
        ! The values of the From:, To:, Date:, and Subject: fields
        ! are parsed because they are used for final formatting of
        ! the introduction text. These fields also often appear as
        ! part of the original message appended to a new message -
        ! and some clients even use localized field names.
        :FROM "From:" .
        :TO "To:" .
        :DATE "Date:", "Sent:" .
        :SUBJECT "Subject:" .

        ! These are some other common (MIME) header fields.
        ! This list needs only to be extended if header detection
        ! should fail.
        :OTHERS
            "Received:", "Cc:", "Bcc:", "Sender:", "Reply-To:",
            "MIME-Version:", "Content-Type:"
        .
    .

    !----------------------------------------------------------------
    ! Quotations from previous messages

    :QUOTES
        ! All these strings may introduce a quoted line.
        :PREFIXES ">", "|", ": " .

        ! To properly detect quoted lines, the preprocessor tries
        ! to fix broken quotes caused by some e-mail clients. Set
        ! this value to 0 to disable this feature.
        :FIX 1
    .

    !----------------------------------------------------------------
    ! Original message separator lines
     
    :ORIGMSG
        ! Delimiters for full quotes (bad style but commonly used)
        ! look something like that:
        !     <line> <keyword> <line>
        ! e.g.
        !     --- Original Message ---
        !       
        ! You can configure minimal line length, legal "line" chars
        ! and keywords.
        !
        ! NOTE: keywords are not case sensitive.

        :MINLINELEN 2
        :LINECHARS "-*#"
        :KEYWORDS
            "Original Message",
            "Reply Separator",
            "Original-Nachricht",
            "UrsprÃ¼ngliche Nachricht",
            "Originalnachricht"
        .   
    .

    !----------------------------------------------------------------
    ! Itemized lists
     
    :LISTS
        ! Lists are detected by looking for bullet characters.
        :BULLETS
            "- ", "* ", "o ",
            "[] ", ":: ",
            "-- ", "** ", "-> ",
            "1. ","1) ",    ! '1' stands for any digit
            "a. ", "a) "    ! 'a' stands for any letter
        .
    .

    !----------------------------------------------------------------
    ! Headlines

    :HL
        ! Lines ending in one of these characters are not
        ! considered to be headlines.
        :WRONGEND "#&,./:;<>\^|~"
    .

    !----------------------------------------------------------------
    ! Emphasis

    :EMPHASIS
        ! It is common style to denote intonation in e-mail by
        ! enclosing words in asterisks or other characters. You
        ! can define a list of such emphasis character sequences.
        ! Note that because the starting and ending sequence may
        ! be different, you have to specify *pairs* of strings.
        :INDICATORS
            "*", "*",
            "_", "_",
            "/", "/",
            "-", "-",
            ">", "<",
            "!", "!"
        .

        ! These punctuation marks are not allowed inside
        ! emphasized text.
        :PUNCT ".!?,;:"
    .

    !----------------------------------------------------------------
    ! The simple (prefix based) URL detection is only implemented
    ! to avoid confusion with pcode.

    :URLS
        ! The simple prefix based URL detection is currently only
        ! implemented to avoid confusion with pcode.
        :PREFIXES
            "http://", "https://", "ftp://", "file://",
            "telnet://", "gopher://", "wais://",
            "nntp://", "prospero://"
        .
    .

    ! one of these characters, a period will be inserted.
    :SENTEND ".!?"

    !----------------------------------------------------------------
    ! Header fields
    ! The rendering of the message header is determined by the
    ! subject line prefix (e.g. RE:, FW:, etc): If the subject
    ! line starts with a known prefix, the corresponding format
    ! is used. If the prefix is unknown or if no prefix is found
    ! at all, a default format is used.

    :HEADER

            "fyi:",
            "__empnosubj__",
            "__empsms__"
            
        .

        ! Header rendering format strings. The first format is the
        ! default format. %1 is substituted by the originator of the
        ! message, %2 and %3 are substituted by the date and time
        ! resp. the message was sent, and %4 is substituted by the
        ! subject line (without any prefixes).
        !
        ! prompt concepts for texts below are as follows:
        ! EPP_HEADER_DEFAULT_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_REPLY_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_FORWARD_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_URGENT_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_REQUEST_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_FYI_READOUT_From_All_Info_PROMPT
	! EPP_HEADER_DEFAULT_READOUT_From_Only_Date_Time_PROMPT
